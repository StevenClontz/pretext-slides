\begin{definition}{}{definition-19}%
\hypertarget{p-21}{}%
Let \(X\) be a set, and let \(\mc T\subseteq \mc P(X)\) satisfy the following properties.%
\leavevmode%
\begin{enumerate}
\item\hypertarget{li-25}{}\(\emptyset,X\in\mc T\).%
\item\hypertarget{li-27}{}If \(\mc U\subseteq\mc T\), then \(\bigcup\mc U\in\mc T\).%
\item\hypertarget{li-30}{}If \(U,V\in\mc T\), then \(U\cap V\in\mc T\).%
\end{enumerate}
\hypertarget{p-33}{}%

(Usually \(\tuple{X,\mc T}\) is abbreviated to just \(X\) when the topology is known from context.)%
\end{definition}
